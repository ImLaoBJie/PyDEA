import matplotlib.pyplot as plt
import pandas as pd

from Pyfrontier.frontier_model import MultipleDEA

sample_df = pd.DataFrame(
    {
        "input": [
            1,
            2,
            4,
            6,
        ],
        "output": [
            0.5,
            2,
            4,
            5,
        ],
    }
)

dea = MultipleDEA("VRS", "in")
dea.fit(
    sample_df[["input"]].to_numpy(),
    sample_df[["output"]].to_numpy(),
)
